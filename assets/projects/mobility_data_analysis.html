def clean_google(df):
    df_google_mobility = df[df['sub_region_1'] == 'California']
    df_google_mobility.drop(df_google_mobility.columns[[0,1,2,4,5,6]], axis=1, inplace=True)

    df_google_mobility.reset_index(drop=True, inplace=True)

    return df_google_mobility

def get_last_14_days_avg(df, date_use):
    date_2 = pd.to_datetime(date_use)
    date_1 = pd.to_datetime(pd.to_datetime(date_use) - timedelta(days=14), format='%Y%m%d')
    df_new = df[(df['date'] >= date_1) & (df['date'] <= date_2)]
    df_grouped = df_new.groupby('county', as_index=False)['retail_and_recreation_percent_change_from_baseline',
                             'grocery_and_pharmacy_percent_change_from_baseline',
                             'parks_percent_change_from_baseline',
                             'transit_stations_percent_change_from_baseline',
                             'workplaces_percent_change_from_baseline',
                             'residential_percent_change_from_baseline'].mean()
    return df_grouped

def get_apple_mobility(mobility_type, df, date_use):
    date_2 = pd.to_datetime(date_use)
    date_1 = pd.to_datetime(pd.to_datetime(date_use) - timedelta(days=14), format='%Y%m%d')
    mob_ca = df[df['transportation_type'] == mobility_type]
    mob_ca = mob_ca[(mob_ca['date'] >= date_1) & (mob_ca['date'] <= date_2)]
    mob_ca = mob_ca.groupby('county', as_index=False)['value'].mean()
    mob_ca = mob_ca.rename(columns={"value":mobility_type})
    return mob_ca
