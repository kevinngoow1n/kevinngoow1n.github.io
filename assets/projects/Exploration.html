# policy at enact date
def draw_policies(year, month, day):
    start_date = datetime.date(year, month, day)
    next_to_draw, next_width = None, 0.5
    for date in df_policy[df_policy['Mandate']==1]['DateEnacted']:
        curr_date = datetime.date(date//10000, date%10000//100, date%100)
        delta = curr_date - start_date
        plt.axvline(x=delta.days, color='brown', linewidth=1)

    for date in df_policy[df_policy['Mandate']==0]['DateEnacted']:
        curr_date = datetime.date(date//10000, date%10000//100, date%100)
        delta = curr_date - start_date
        plt.axvline(x=delta.days, color='salmon', ls='--', linewidth=1)

# Google Mobility & COVID
X_y = google_mobility_covid.dropna()
prevs = [X_y['new_cases'].iloc[i-1] if i > 0 else 0 for i in range(X_y.shape[0])]
X = X_y [['retail_and_recreation', 'grocery_and_pharmacy', 'parks', 'transit_stations', 'workplaces', 'residential']]
X = X.assign(prev = prevs)
y = X_y['new_cases']
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

regressor = LinearRegression()
regressor.fit(X_train, y_train)

y_pred = regressor.predict(X_test)
print('Mean Number of New Cases:', google_mobility_covid['new_cases'].mean())
print('Mean Absolute Error:', metrics.mean_absolute_error(y_test, y_pred))
print('Mean Squared Error:', np.sqrt(metrics.mean_squared_error(y_test, y_pred)))
